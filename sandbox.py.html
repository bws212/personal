45   def residuals_safe(params, T_exp, P_exp):
46       A, B, C = params
47       # Ensure T + C remains positive and above a small threshold (e.g., 1) to avoid division issues
48       T_adjusted = T_exp + np.abs(C) + 1
49       P_pred = 10 ** (A - B / T_adjusted)
50       return np.log10(P_pred) - np.log10(P_exp)
